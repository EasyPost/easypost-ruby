def estimate_delivery_date(params = {})
  url = 'smartrate/deliver_by'

  response = @client.make_request(:post, url, params)
  EasyPost::InternalUtilities::Json.convert_json_to_object(response, EasyPost::Models::EasyPostObject)
end
