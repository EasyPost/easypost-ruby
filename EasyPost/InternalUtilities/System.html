def self.os_name
  case RUBY_PLATFORM
  when /linux/i
    'Linux'
  when /darwin/i
    'Darwin'
  when /cygwin|mswin|mingw|bccwin|wince|emx/i
    'Windows'
  else
    'Unknown'
  end
end
